\begin{slide}{Temas e Objetivos}

	\begin{objectiveBox}[\linewidth]
	\justifying
    	\textbf{Objetivo Principal:}
    	\begin{Item}
    	    \item Avaliar estratégias de modelagem baseada em dados e híbrida físico-dados para predição de desempenho em destilação por membranas V-AGMD.
    	    \item Predizer fluxo de permeado e temperaturas de saída a partir de variáveis operacionais.
        \end{Item}
	\end{objectiveBox}
	
\end{slide}

\begin{slide}{Contribuições do Trabalho}

    \begin{objectiveBox}[\linewidth]
    \justifying
    \textbf{Contribuições:} 
        \begin{Item}
    	    \item Pipeline unificado com GroupKFold por regimes experimentais e critério hierárquico de seleção (1-SE + complexidade).
    	    \item Implementação e comparação de arquiteturas híbridas que fusionam conhecimento físico com modelos baseados em dados: HPD, Residual, HRNN e Luc.
    	    \item Identificação de trade-off entre predição de fluxo (híbridos) e temperaturas (OLS), com recomendação de modelo distinto para cada variável.
        \end{Item}
    \end{objectiveBox}	
	
\end{slide}
