\begin{slide}{AnĂĄlise de CorrelaĂ§ĂŁo}
  \small
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Coeficiente de Pearson ($r$)}
      \begin{Item}[0.15]
        \item Mede a relaĂ§ĂŁo \textbf{linear} entre duas variĂĄveis
        \item $r \in [-1, 1]$: $|r|$ prĂłximo de 1 indica forte correlaĂ§ĂŁo linear
        \item SensĂ­vel a outliers
        \item $r = \frac{\sum (x_i - \bar{x})(y_i - \bar{y})}{\sqrt{\sum (x_i - \bar{x})^2 \sum (y_i - \bar{y})^2}}$
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Coeficiente de Spearman ($\rho$)}
      \begin{Item}[0.15]
        \item Mede a relaĂ§ĂŁo \textbf{monotĂ´nica} (baseada em postos)
        \item $\rho \in [-1, 1]$: captura tendĂŞncias crescentes ou decrescentes
        \item Menos sensĂ­vel a outliers que Pearson
        \item Calculado sobre os \textit{ranks} (postos) dos dados
      \end{Item}
    \end{column}
  \end{columns}
  \vfill
  \begin{Item}
    \item \textbf{Papel no ML:} identificar associaĂ§Ăľes entre variĂĄveis; orientar seleĂ§ĂŁo de atributos; detectar multicolinearidade entre preditores
    \item Pearson $\approx$ Spearman $\rightarrow$ relaĂ§ĂŁo aproximadamente linear
    \item Pearson $\neq$ Spearman $\rightarrow$ relaĂ§ĂŁo nĂŁo linear ou presenĂ§a de outliers
  \end{Item}
  \note{A anĂĄlise de correlaĂ§ĂŁo ĂŠ uma ferramenta exploratĂłria fundamental em ML. O coeficiente de Pearson mede associaĂ§Ăľes lineares, enquanto o de Spearman captura tendĂŞncias monotĂ´nicas mais gerais. A comparaĂ§ĂŁo entre eles ajuda a identificar a natureza das relaĂ§Ăľes entre variĂĄveis, orientando a escolha de modelos e a seleĂ§ĂŁo de atributos. [~1,5 min]}
\end{slide}

\begin{slide}{SeleĂ§ĂŁo de Atributos e DiscretizaĂ§ĂŁo}
