\begin{slide}{Conclusões}
  \small
  \centering
  \resizebox{\textwidth}{!}{
    \begin{tabular}{p{0.2\linewidth} p{0.35\linewidth} p{0.35\linewidth}}
      \toprule
      \textbf{Aspecto} & \textbf{Resultado} & \textbf{Implicação} \\
      \midrule
      \textbf{Melhor modelo}
      & ZohanResidual: RMSE 0,061 (fluxo), redução de 31\% vs OLS (0,089)
      & Arquitetura residual é a mais eficaz para predição do fluxo \\
      \midrule
      \textbf{Modelos lineares}
      & OLS: melhor para T$_{out}$ (RMSE 0,246 e 0,251)
      & Comportamento térmico é aproximadamente linear na faixa estudada \\
      \midrule
      \textbf{Validação}
      & GroupKFold por regimes evitou vazamento; critério 1-SE + complexidade
      & Pipeline mais robusto que a literatura (que usa hold-out simples) \\
      \midrule
      \textbf{Hibridização}
      & 4 arquiteturas implementadas; residuais superaram regularização na perda
      & Estratégia de correção residual é a mais promissora \\
      \midrule
      \textbf{Limitação}
      & Modelo híbrido piora desempenho nas temperaturas em relação ao OLS
      & Sugere seleção multi-objetivo ou modelos distintos por variável \\
      \bottomrule
    \end{tabular}
  }
  \note{Conclusões principais: ZohanResidual venceu para fluxo com RMSE 0,061; OLS é melhor para temperaturas; o pipeline com GroupKFold é mais robusto que a literatura; arquiteturas residuais são mais eficazes que regularização na perda. [~1,5 min]}
\end{slide}
