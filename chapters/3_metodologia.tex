\begin{slide}{Validação Cruzada e Seleção de Modelos}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{GroupKFold por Regime}
      \begin{Item}
        \item K=3 partições: um regime reservado p/ validação
        \item Grupo = Regime (10, 40, 70 g/L)
        \item Evita vazamento entre amostras correlacionadas
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Critério de seleção (1-SE)}
      \begin{Item}
        \item 1º: melhor RMSE$_{CV}$ no fluxo
        \item 2º: modelos dentro de 1 desvio padrão do melhor
        \item 3º: menor complexidade entre eles
      \end{Item}
    \end{column}
  \end{columns}
  \vfill
  \begin{myBox}
    \textbf{Métrica final:} RMSE e R² sobre predições \textit{Out-of-Fold} (OOF) — modelo retreinado do zero em cada partição
  \end{myBox}
  \note{A validação usa GroupKFold com 3 partições, separando por regime de salinidade. O critério de seleção é hierárquico: melhor RMSE no fluxo, modelos dentro de 1 desvio padrão, e menor complexidade. A métrica final é calculada sobre predições out-of-fold. [~1,5 min]}
\end{slide}

\begin{slide}{Arquiteturas Híbridas}
  \small
  \centering
  \resizebox{\textwidth}{!}{
    \begin{tabular}{p{0.2\linewidth} p{0.3\linewidth} p{0.2\linewidth} p{0.2\linewidth}}
      \toprule
      \textbf{Arquitetura} & \textbf{Como integra física + dados} & \textbf{Pontos fortes} & \textbf{Limitações} \\
      \midrule
      \textbf{FrozenBaseline}
      & Baseline neural pura (sem física)
      & Referência data-driven justa
      & Não usa física \\
      \midrule
      \textbf{HPD}
      & Entrada expandida: $[X, Y_{phy}]$ — física como feature extra
      & Simples; rede explora correlações
      & Pode ignorar física \\
      \midrule
      \textbf{ZohanResidual}
      & $Y_{res} = Y - Y_{phy}$; rede aprende correção residual
      & Melhor RMSE para fluxo
      & Performance nas T$_{out}$ inferior ao linear \\
      \midrule
      \textbf{HRNN}
      & Combina HPD + Residual: entrada expandida + correção
      & Mais flexível
      & Maior complexidade \\
      \midrule
      \textbf{Luc}
      & Regularização física na função de perda
      & Preserva consistência
      & Menor ganho prático \\
      \bottomrule
    \end{tabular}
  }
  \note{Quatro arquiteturas híbridas foram implementadas. A FrozenBaseline é a referência neural sem física. A HPD adiciona a física como feature. A ZohanResidual aprende a correção residual. A HRNN combina ambas. A Luc usa regularização na perda. [~1,5 min]}
\end{slide}
