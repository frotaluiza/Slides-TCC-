\section{Revisão Bibliográfica}

\begin{slide}{Abordagens de Modelagem em MD}
  \small
  \centering
  \resizebox{\textwidth}{!}{
    \begin{tabular}{p{0.18\linewidth} p{0.15\linewidth} p{0.32\linewidth} p{0.15\linewidth} p{0.15\linewidth}}
      \toprule
      \textbf{Abordagem} & \textbf{Tecnologia} & \textbf{Descrição} & \textbf{Vantagens} & \textbf{Limitações} \\
      \midrule
      \textbf{Física}
      & AGMD, AGMD (CFD), \textbf{V-AGMD}
      & Modelos distribuídos (Alsaadi, 2013; Ansari, 2022) e reduzido 0D (\textbf{Lisboa}, 2024)
      & Interpretabilidade física
      & Alto custo; simplificações \\
      \midrule
      \textbf{Data-driven}
      & \textbf{AGMD}, VMD
      & ANNs p/ predição de fluxo e temperaturas (\textbf{Ibnouf} et al.; \textbf{Mittal} et al.)
      & Alta flexibilidade; aprox. não linear
      & Generalização limitada \\
      \midrule
      \textbf{Híbridos}
      & MD (geral), SGMD, DCMD
      & PINNs na função de perda (Andrés-Mañas; Shirzadi, 2024); FO-PINNs (Shahouni, 2026)
      & Combina física + dados
      & Complexidade de implementação \\
      \bottomrule
    \end{tabular}
  }
  \vfill
  \textbf{Lacuna:} nenhum estudo aplica PINNs especificamente a AGMD ou V-AGMD até o momento
  \note{A revisão mostra três abordagens com as respectivas tecnologias de MD. Física: AGMD e V-AGMD. Data-driven: AGMD e VMD. Híbridos: SGMD e DCMD, mas nenhum para AGMD ou V-AGMD — essa é a lacuna que motiva este trabalho. [~1,5 min]}
\end{slide}

\begin{slide}{Lacunas e Contribuições}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Lacunas na literatura:}
      \begin{Item}
        \item Divisão aleatória simples (70/15/15 ou 80/20)
        \item Validação cruzada raramente usada
        \item Seleção de hiperparâmetros heurística
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Contribuições do trabalho:}
      \begin{Item}
        \item GroupKFold por regimes experimentais
        \item Busca sistemática de hiperparâmetros
        \item Critério 1-SE + menor complexidade
        \item 4 arquiteturas híbridas implementadas
      \end{Item}
    \end{column}
  \end{columns}
  \note{A literatura tem limitações em validação e seleção. Este trabalho contribui com validação por grupos, busca sistemática, e critério de complexidade. [~1 min]}
\end{slide}
