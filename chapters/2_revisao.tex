\section{Revisão Bibliográfica}

\begin{slide}{Desafios e Abordagens de Modelagem em MD}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Desafios:}
      \begin{Item}
        \item Fenômenos acoplados de calor e massa
        \item Múltiplas variáveis: T, vazão, pressão, salinidade
        \item Polarização térmica, perdas de calor, escala piloto
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Três abordagens:}
      \begin{Item}
        \item \textbf{Físicos:} balanços de massa/energia, alto custo (Ansari, 2022)
        \item \textbf{Data-driven:} ANNs mapeiam entradas/saídas, mas generalização limitada (Ibnouf et al.)
        \item \textbf{Híbridos:} PINNs + residuais combinam física e dados (Zhou et al., 2024)
      \end{Item}
    \end{column}
  \end{columns}
  \note{A modelagem de MD enfrenta desafios como fenômenos acoplados e múltiplas variáveis. Três abordagens principais: modelos físicos (precisos mas caros), data-driven (flexíveis mas dependentes dos dados), e híbridos que combinam o melhor dos dois mundos. [~1 min]}
\end{slide}
%========================================================================

\begin{slide}{Lacunas da Literatura e Contribuições}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Lacunas identificadas:}
      \begin{Item}
        \item Divisão aleatória simples dos dados (70/15/15 ou 80/20)
        \item Validação cruzada raramente empregada
        \item Seleção de hiperparâmetros heurística
        \item Avaliação apenas em interpolação
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Este trabalho:}
      \begin{Item}
        \item GroupKFold por regimes experimentais
        \item Busca sistemática de hiperparâmetros
        \item Critério 1-SE + menor complexidade
        \item 4 arquiteturas híbridas implementadas
      \end{Item}
    \end{column}
  \end{columns}
  \note{A literatura tem limitações claras: divisão simples dos dados sem validação cruzada, seleção heurística, e foco só em interpolação. Este trabalho contribui com validação por grupos experimentais, busca sistemática, e um critério de seleção que penaliza complexidade. [~1 min]}
\end{slide}
