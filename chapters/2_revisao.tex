\section{Revisão Bibliográfica}

\begin{slide}{Abordagens de Modelagem em MD}
  \small
  \centering
  \resizebox{\textwidth}{!}{
    \begin{tabular}{p{0.2\linewidth} p{0.25\linewidth} p{0.25\linewidth} p{0.2\linewidth}}
      \toprule
      \textbf{Abordagem} & \textbf{Descrição} & \textbf{Vantagens} & \textbf{Limitações} \\
      \midrule
      \textbf{Física (1\textordfeminine{} princípios)}
      & Modelos distribuídos (Alsaadi, 2013; Ansari, 2022) e reduzido 0D (Lisboa, 2024)
      & Interpretabilidade física; consistência termodinâmica
      & Alto custo computacional; simplificações estruturais \\
      \midrule
      \textbf{Data-driven (ANN)}
      & ANNs p/ predição de fluxo e temperaturas (Ibnouf et al.; Mittal et al.)
      & Alta flexibilidade; aproximação não linear
      & Generalização limitada; dependente da distribuição dos dados \\
      \midrule
      \textbf{Híbridos}
      & PINNs: equações governantes na função de perda (Andrés-Mañas et al.)
      & Combina física + dados; melhor consistência
      & Maior complexidade de implementação \\
      \bottomrule
    \end{tabular}
  }
  \vfill
  \textbf{Desafio central:} fenômenos acoplados calor/massa, múltiplas variáveis (T, vazão, P, salinidade), dados experimentais limitados
  \note{A revisão mostra três grandes abordagens. Modelos físicos são interpretáveis mas caros. Data-driven são flexíveis mas dependem dos dados. Os híbridos combinam o melhor dos dois mundos. O desafio central é modelar sistemas com fenômenos acoplados e dados limitados. [~1,5 min]}
\end{slide}

\begin{slide}{Lacunas e Contribuições}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Lacunas na literatura:}
      \begin{Item}
        \item Divisão aleatória simples (70/15/15 ou 80/20)
        \item Validação cruzada raramente usada
        \item Seleção de hiperparâmetros heurística
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Contribuições do trabalho:}
      \begin{Item}
        \item GroupKFold por regimes experimentais
        \item Busca sistemática de hiperparâmetros
        \item Critério 1-SE + menor complexidade
        \item 4 arquiteturas híbridas implementadas
      \end{Item}
    \end{column}
  \end{columns}
  \note{A literatura tem limitações em validação e seleção. Este trabalho contribui com validação por grupos, busca sistemática, e critério de complexidade. [~1 min]}
