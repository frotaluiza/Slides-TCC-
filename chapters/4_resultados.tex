\begin{slide}{Etapa 2.2: Comparação entre Estratégias de Validação}
  \small
  \begin{columns}
    \begin{column}{0.48\linewidth}
      \textbf{Objetivo:} justificar o uso do GroupKFold comparando com KFold comum
      \vspace{0.2cm}
      \begin{Item}[0.15]
        \item Modelos do Stage 0 treinados com ambas as estratégias
        \item RMSE calculado sob cada tipo de partição
        \item KFold comum: amostras do mesmo regime podem estar em treino e validação
        \item GroupKFold: cada fold treina 2 regimes, valida o 3°
      \end{Item}
    \end{column}
    \begin{column}{0.48\linewidth}
      \textbf{Diferença observada — RMSE Flux (Stage 0)}
      \scriptsize
      \begin{tabular}{lcc}
        \toprule
        \textbf{Modelo} & \textbf{CV KFold} & \textbf{CV GroupKFold} \\
        \midrule
        OLS            & 0.078 & 0.081 \\
        Ridge (Ref)    & 0.078 & 0.082 \\
        Lasso\_Indep   & 0.079 & 0.082 \\
        DT (Alim)      & 0.269 & \textbf{0.485} \\
        RF (Ref)       & 0.199 & \textbf{0.422} \\
        GB (Alim)      & 0.085 & \textbf{0.358} \\
        \bottomrule
      \end{tabular}
      \vspace{0.2cm}
      \begin{tcolorbox}[colback=orange!10, colframe=orange!40, arc=2pt, boxrule=0.5pt]
        \scriptsize \textbf{Conclusão:} O uso do GroupKFold proporciona uma boa medição do erro de generalização do modelo, evitando que a avaliação seja influenciada pela correlação entre amostras de um mesmo regime experimental.
      \end{tcolorbox}
    \end{column}
  \end{columns}
  \note{A comparação entre KFold comum e GroupKFold mostra diferenças mensuráveis na performance de todos os modelos do Stage 0, confirmando que a estratégia de validação impacta a estimativa de erro. O GroupKFold foi adotado por respeitar a estrutura de grupos dos dados experimentais. [~1,5 min]}
\end{slide}

\begin{slide}{Teste Hold-Out — Valida\c{c}\~ao Final dos Modelos}
  \small
  \begin{columns}
    \begin{column}{0.48\linewidth}
      \textbf{O que \'e o hold-out?}
      \vspace{0.2cm}
      \begin{Item}[0.15]
        \item 20\% dos dados (35 amostras) separados \textbf{antes} de qualquer modelagem
        \item Cont\'em amostras dos 3 regimes de salinidade
        \item Nunca \'e usado durante treinamento, valida\c{c}\~ao ou sele\c{c}\~ao de hiperpar\^ametros
        \item Fornece uma estimativa realista do desempenho em condi\c{c}\~oes n\~ao vistas
      \end{Item}
      \vspace{0.3cm}
      \textbf{Por que \'e necess\'ario?}
      \begin{Item}[0.15]
        \item O GroupKFold avalia generaliza\c{c}\~ao entre regimes, mas a m\'edia dos 3 folds ainda pode ter vi\'es de sele\c{c}\~ao
        \item O hold-out \'e a \'ultima barreira: se o modelo vai bem aqui, a confian\c{c}a no pipeline \'e maior
      \end{Item}
    \end{column}
    \begin{column}{0.48\linewidth}
      \textbf{Resultados consolidados}
      \vspace{0.2cm}
      \centering
      \begin{tabular}{lccc}
        \toprule
        \textbf{Vari\'avel} & \textbf{Modelo} & \textbf{Test RMSE} & \textbf{vs 0D} \\
        \midrule
        T$_{alim,out}$ & PhyResidual (Alim) & 0,225 & $-$85\% \\
        T$_{ref,out}$  & PhyHybrid (Alim)   & 0,230 & $-$73\% \\
        Flux           & PhyResidual (Alim) & \textbf{0,057} & $-$72\% \\
        \midrule
        \multicolumn{4}{l}{\scriptsize Ref. 0D: Talim=1,549 \; Tref=0,851 \; Flux=0,206} \\
        \bottomrule
      \end{tabular}
      \vspace{0.2cm}
      \begin{tcolorbox}[colback=blue!6, colframe=blue!40, arc=2pt, boxrule=0.5pt]
        \scriptsize \textbf{Observa\c{c}\~oes:}
        \begin{Item}[0.05]
          \item T$_{alim,out}$: Stage 1 teve melhor RMSE (0,215), mas PhyResidual (0,225) chega pr\'oximo com R$^2$ 0,981.
          \item T$_{ref,out}$: Ridge (Stage 0, 0,219) superou as h\'ibridas — a fenomenologia linear j\'a captura bem esta vari\'avel.
          \item Flux: PhyResidual (Alim) selecionado por generaliza\c{c}\~ao mais est\'avel (Dif $-$5,0\% vs $-$25,0\% da vers\~ao Flux), com RMSE 0,057.
        \end{Item}
      \end{tcolorbox}
    \end{column}
  \end{columns}
  \note{O hold-out de 20\% \'e uma etapa adicional de valida\c{c}\~ao para garantir que os modelos n\~ao est\~ao superestimando o desempenho devido \'a correla\c{c}\~ao entre amostras dentro dos folds. [~1,5 min]}
\end{slide}
