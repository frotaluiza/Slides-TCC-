\begin{slide}{Ilha de Policogeração Sustentável (LabMEMS/COPPE)}
  \begin{columns}
    \begin{column}{.55\linewidth}
      \begin{Item}
        \item Protótipo pioneiro: água potável + energia elétrica via solar térmica
        \item 5 kW$_e$ nominais ($\approx$ 25 residências) + 8 kW$_t$ recuperáveis
        \item Dessalinizador V-AGMD: 1.000 L/dia ($>$ 100 pessoas/dia)
        \item Foco: comunidades remotas/off-grid (Semiárido, ilhas, O\&G)
        \item Alinhado aos ODS 6 (água), ODS 7 (energia limpa) e ODS 13 (clima)
      \end{Item}
    \end{column}
    \begin{column}{.45\linewidth}
      \includegraphics[width=\linewidth]{tcc_images/instalacao_ips.png}
    \end{column}
  \end{columns}
  \vfill\hfill{\scriptsize Fontes: COPPE/UFRJ, 2022; LISBOA et al., 2024; Imagem: SEIXO, 2024}
  \note{O projeto está inserido na Ilha de Policogeração Sustentável da COPPE, inaugurada em 2022. É um sistema pioneiro que integra painéis solares de alta concentração com um dessalinizador V-AGMD, produzindo água para mais de 100 pessoas por dia. O foco são comunidades remotas sem acesso à rede elétrica. [~1 min]}
\end{slide}

\begin{slide}{V-AGMD: Inovação Tecnológica}
  \begin{columns}
    \begin{column}{.55\linewidth}
      \begin{Item}
        \item \textbf{MD:} processo híbrido térmico-membrana — vapor por gradiente térmico (Cath, 2010)
        \item \textbf{Diferenciais:} baixa T/P, $>$99,9\% rejeição, trata salmouras (Yadav et al., 2021)
        \item \textbf{V-AGMD (adotada):} lacuna de ar + vácuo parcial $\rightarrow$ maior fluxo (Warsinger, 2017)
        \item \textbf{Inovação:} opera com calor solar/residual, sem alta pressão
        \item \textbf{Desafio:} modelagem preditiva com dados experimentais limitados
      \end{Item}
    \end{column}
    \begin{column}{.45\linewidth}
      \includegraphics[width=\linewidth]{tcc_images/distillateur_vagmd.png}
    \end{column}
  \end{columns}
  \vfill\hfill{\scriptsize Fontes: CATH, 2010; WARSINGER, 2017; YADAV et al., 2021; Imagem: SEIXO, 2024}
  \note{A destilação por membrana é a tecnologia central. Ela combina princípios térmicos com membrana: o vapor passa por uma membrana hidrofóbica impulsionado por calor. A configuração V-AGMD adiciona vácuo na lacuna de ar para aumentar o fluxo. A grande inovação é poder operar com calor solar ou residual, sem precisar de altas pressões como a osmose reversa. O desafio é modelar esse sistema com dados limitados. [~1 min]}
\end{slide}
%========================================================================

\begin{slide}{Objetivos}
  \begin{Item}
    \item \textbf{Geral:} avaliar modelagem baseada em dados e híbrida para predição do desempenho V-AGMD
    \item \textbf{Específicos:}
    \begin{Item}
      \item Estruturar dados experimentais com separação por regimes
      \item Comparar 4 famílias: lineares, árvores, redes neurais e híbridas
      \item Aplicar validação cruzada por grupos (GroupKFold)
      \item Selecionar modelo por desempenho, generalização e complexidade
    \end{Item}
  \end{Item}
  \note{O objetivo geral é avaliar diferentes estratégias de modelagem — desde regressões lineares até arquiteturas híbridas — para prever o desempenho do dessalinizador V-AGMD. Os objetivos específicos guiam o pipeline metodológico. [~1 min]}
\end{slide}
