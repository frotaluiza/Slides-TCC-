\begin{slide}{A Crise Global da Água}
  \begin{Item}
    \item 4 bilhões de pessoas enfrentam escassez severa $\ge$1 mês/ano (Mekonnen \& Hoekstra, 2016)
    \item 2,2 bilhões sem acesso a água potável segura (WHO, 2023)
    \item Demanda global deve superar oferta em 40\% até 2030 (IDRA)
    \item Até 2050: $\frac{3}{4}$ da população podem sofrer impactos de seca (UNU-INWEH)
  \end{Item}
  \note{Abro com números impactantes da ONU: 4 bilhões de pessoas sofrem com escassez hídrica severa, 2,2 bilhões não têm água potável segura, e a demanda deve superar a oferta em 40\% até 2030. Esse é o problema que motiva o trabalho. [~1 min]}
\end{slide}

\begin{slide}{Dessalinização: Solução Global e Brasileira}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \begin{Item}
        \item Única fonte ``ilimitada'' de água potável (97,5\% da Terra é água salgada)
        \item ODS 6 (Target 6.A): dessalinização como tecnologia para água potável
        \item Mercado: US\$ 7,1B $\rightarrow$ US\$ 11,2B (CAGR 9,5\%)
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \begin{Item}
        \item Brasil: pior seca da história (2010--2019)
        \item Cantareira (SP) ainda em atenção (2026)
        \item 8.500 km de costa $\rightarrow$ enorme potencial
      \end{Item}
    \end{column}
  \end{columns}
  \note{A dessalinização é a única forma de criar água doce nova. O ODS 6 da ONU trata de água potável e saneamento para todos, e o Target 6.A cita explicitamente a dessalinização como tecnologia-chave para cooperação internacional. Já o ODS 7 trata de energia limpa — nosso projeto no LabMEMS integra ambos, usando energia solar para dessalinizar água. O mercado global cresce 9,5\% ao ano. No Brasil, temos 8.500 km de costa e um histórico de secas severas — o Cantareira ainda está em atenção. [~1 min]}
\end{slide}

\begin{slide}{Tecnologias e o Diferencial da MD}
  \begin{columns}
    \begin{column}{.5\linewidth}
      \begin{Item}
        \item \textbf{Térmicas:} MED, MSF (evaporação, alta energia)
        \item \textbf{Membrana:} RO domina mercado, mas exige alta pressão e pré-tratamento
        \item \textbf{MD (adotada):} processo híbrido térmico-membrana
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \begin{figure}[H]
        \includegraphics[width=0.9\linewidth]{tcc_images/chap01/membrane.png}
      \end{figure}
    \end{column}
  \end{columns}
  \vfill
  \begin{myBox}
    \textbf{Vantagens da MD:} baixa T/P, $>$99,9\% rejeição, trata salmouras, calor solar/residual
  \end{myBox}
  \note{Comparo as tecnologias: RO domina mas tem limitações. A MD combina o melhor dos dois mundos — opera a baixas temperaturas e pressões, rejeita mais de 99,9\% dos sais, e pode usar calor residual ou solar. A configuração V-AGMD adiciona vácuo para aumentar o fluxo. [~1 min]}
\end{slide}

\begin{slide}{Motivação e Contexto do Projeto}
  \begin{Item}
    \item Projeto multidisciplinar LabMEMS/UFRJ: cogeração água (ODS 6) + energia (ODS 7) via solar térmica
    \item Desafio: prever fluxo de permeado e temperaturas de saída da V-AGMD
    \item Dados limitados (10, 40, 70 g/L) $\rightarrow$ generalização é desafio central
    \item Abordagem: modelo físico 0D + dados experimentais $\rightarrow$ estratégias híbridas
  \end{Item}
  \vfill\hfill{\scriptsize LISBOA et al., 2024}
  \note{O projeto está inserido no LabMEMS/UFRJ, onde temos uma planta de cogeração solar. O desafio é modelar o desempenho com dados experimentais limitados — apenas três regimes de salinidade. A solução proposta combina um modelo físico com aprendizado de máquina. [~1 min]}
\end{slide}
%========================================================================

\begin{slide}{Objetivos}
  \begin{Item}
    \item \textbf{Geral:} avaliar modelagem baseada em dados e híbrida para predição do desempenho V-AGMD
    \item \textbf{Específicos:}
    \begin{Item}
      \item Estruturar dados experimentais com separação por regimes
      \item Comparar 4 famílias: lineares, árvores, redes neurais e híbridas
      \item Aplicar validação cruzada por grupos (GroupKFold)
      \item Selecionar modelo por desempenho, generalização e complexidade
    \end{Item}
  \end{Item}
  \note{O objetivo geral é avaliar diferentes estratégias de modelagem — desde regressões lineares até arquiteturas híbridas — para prever o desempenho do dessalinizador V-AGMD. Os objetivos específicos guiam o pipeline. [~1 min]}
\end{slide}
