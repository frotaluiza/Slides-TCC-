\begin{slide}{Etapa 3: Critérios de Seleção e Métricas de Erro}
  \small
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Critério hierárquico:}
      \begin{Item}[0.15]
        \item \textbf{Nível 1 — Dentro de cada família:} 1-SE + mínima complexidade para escolher hiperparâmetros
        \item \textbf{Nível 2 — Entre famílias:} comparação direta pelo menor RMSE$_{Test}$ no fluxo
      \end{Item}
      \textbf{Métricas de erro}
      \begin{Item}[0.15]
        \item \textbf{Erro de generalização:} média do RMSE entre os 3 folds do GroupKFold — guia a busca de hiperparâmetros e seleção dentro de cada família
        \item \textbf{Erro de teste (hold-out):} RMSE calculado no hold-out de 35 amostras — métrica final para comparar modelos selecionados
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Seleção por target:}
      \begin{Item}[0.15]
        \item Cada variável de saída (Flux, T$_{alim,out}$, T$_{ref,out}$) pode ter um modelo diferente
        \item A busca de hiperparâmetros é otimizada separadamente para cada alvo
        \item Ao final, cada target tem seu próprio modelo selecionado pelo critério 1-SE
      \end{Item}
      \vspace{0.3cm}
      \textbf{Hold-out:}
      \begin{Item}[0.15]
        \item 20\% dos dados (35 amostras) separado \textbf{antes} de qualquer modelagem
        \item Contém amostras dos 3 regimes de salinidade
        \item O RMSE no hold-out estima o desempenho em condições operacionais representativas
      \end{Item}
    \end{column}
  \end{columns}
\end{slide}
\begin{slide}{Etapa 4: Análise Exploratória}
  \small
  \begin{Item}
    \item Objetivo: identificar padrões, tendências e relações entre variáveis antes da modelagem
    \item Visualização por gráficos de dispersão entre entradas e saídas
    \item Análise de correlação para quantificar associações entre variáveis
    \item Identificação de variáveis com maior influência sobre cada saída
  \end{Item}
\end{slide}
\begin{slide}{Etapa 5: Modelo Físico Reduzido 0D}
  \small
  \begin{Item}[0.15]
    \item \textbf{Referência física}: modelo fenomenológico do módulo V-AGMD baseado em balanços de massa e energia — serve como baseline para comparar o ganho dos modelos orientados por dados
    \item \textbf{Fonte de informação para híbridos}: as predições do modelo 0D são incorporadas às arquiteturas híbridas (PhyInput, PhyResidual, PhyHybrid, PhyLoss) como entrada auxiliar ou termo de regularização
    \item \textbf{Implementação}: código em C executado via script Python para cada condição experimental, gerando três colunas preditas (\texttt{phy}) na base de dados
  \end{Item}
\end{slide}

\begin{slide}{Etapa 8: Stage 2 — Modelos Híbridos}
  \small
  \centering
  \resizebox{\textwidth}{!}{
    \begin{tabular}{p{0.2\linewidth} p{0.3\linewidth} p{0.2\linewidth} p{0.2\linewidth}}
      \toprule
      \textbf{Arquitetura} & \textbf{Como integra física + dados} & \textbf{Pontos fortes} & \textbf{Limitações} \\
      \midrule
      \textbf{PhyInput}
      & Entrada expandida: $[X, Y_{phy}]$ & Simples & Pode ignorar física \\
      \midrule
      \textbf{PhyResidual}
      & $Y_{hat} = Y_{phy} + \Delta$ & Melhor RMSE fluxo & T$_{out}$ inferior \\
      \midrule
      \textbf{PhyHybrid}
      & PhyInput + Residual combinados & Mais flexível & Maior complexidade \\
      \midrule
      \textbf{PhyLoss}
      & Regularização física na perda & Consistência & Menor ganho prático \\
      \bottomrule
    \end{tabular}
  }
  \vfill
  \textbf{Busca restrita:} apenas L2, mantendo a baseline fixa — para isolar o efeito da física
\end{slide}
\begin{slide}{Etapa 9: Comparação entre Modelos Selecionados}
  \small
  \begin{columns}
    \begin{column}{.5\linewidth}
      \textbf{Objetivo:}
      \begin{Item}[0.15]
        \item Comparar o desempenho dos modelos selecionados em cada etapa
        \item Confrontar os resultados com o modelo físico 0D (referência fenomenológica)
        \item Verificar se a incorporação de técnicas de ML trouxe ganhos em relação ao modelo 0D isolado
      \end{Item}
      \vspace{0.3cm}
      \textbf{O que é comparado:}
      \begin{Item}[0.15]
        \item Modelos clássicos (OLS, árvores) \textit{vs} redes neurais (MLP)
        \item Redes neurais puras \textit{vs} arquiteturas híbridas (PhyInput, PhyResidual, PhyHybrid, PhyLoss)
        \item Modelos selecionados \textit{vs} modelo físico 0D
      \end{Item}
    \end{column}
    \begin{column}{.5\linewidth}
      \textbf{Métricas de comparação:}
      \begin{Item}[0.15]
        \item RMSE e R² calculados no hold-out para cada variável de saída
        \item Erro de generalização (média do RMSE dos 3 folds do GroupKFold)
      \end{Item}
    \end{column}
  \end{columns}
\end{slide}
